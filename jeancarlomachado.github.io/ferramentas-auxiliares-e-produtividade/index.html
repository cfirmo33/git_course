Funções

current_branch() {
    git branch 2> /dev/null | grep "*" | cut -d" " -f2
}

commit_diff () {
    git diff $1~ $1
}

not_committed_files () {
    git ls-files --others --exclude-standard
}

commit_count () {
    git log --pretty=oneline | wc -l
}


last_diff_file() {
    last_diff=${2:-1}
    file_name=$1
    git log -p --follow -$last_diff $file_name
}
